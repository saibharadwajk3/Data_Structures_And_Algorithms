3)count and say
class Solution {
    public String countAndSay(int n) {
        
        if(n==1) return "1";
        
        StringBuilder sb=new StringBuilder();
        
        String s=countAndSay(n-1);
        
        int len=s.length();
        
        int i=0;
        
        while(i<len){
            
            int count=1;
            char ch=s.charAt(i++);
            
            while(i<len && s.charAt(i)==ch){
                
                ++count;
                ++i;
            }
            sb=sb.append(count).append(ch);
            
        }
        return sb.toString();
        
    }
}
